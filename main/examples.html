SerialÔÉÅ

Serial example demonstrates the usage of serial (UART) communication to receive data from device, parse it and print the data in the terminal. The data coming from the device shold be in the following format:

<data1>,<data2>\n

Example code:

 1import sys
 2import time
 3import queue
 4
 5from pydevdtk.coms.serial import Serial
 6
 7
 8class Parser:
 9    def __init__(self):
10        self.buffer = b""
11
12    def parse_data(self, data):
13        self.buffer += data
14        lines = self.buffer.split(b"\n")
15        self.buffer = lines.pop()
16
17        for line in lines:
18            line = line.strip()
19            if line:
20                values = line.decode("ascii").split(",")
21                try:
22                    parsed_values = tuple(map(float, values))
23                    yield parsed_values
24                except ValueError:
25                    pass
26
27
28ser = Serial()
29com_ports = ser.get_found_devices()
30print(f"Found ports: {com_ports}\n")
31
32port = input("Type in the port to read the data from: ")
33if port not in com_ports:
34    print(f"Requested port {port} not found")
35
36data_queue = queue.Queue()
37opened = ser.open(port, lambda data: data_queue.put(data))
38if not opened:
39    print(f"Error while opening port {port}")
40    sys.exit(1)
41
42parser = Parser()
43t_run = 10
44t_start = time.time()
45while time.time() - t_start < t_run:
46    for sin, cos in parser.parse_data(data_queue.get()):
47        print(f"sin = {sin} | cos = {cos}")
48
49ser.close()

BLEÔÉÅ

BLE example demonstrates the usage of BLE communication to receive data from device, parse the data and print it in the terminal.

The data coming from the device shold be in the following format:

<data1>,<data2>\n

Example code:

 1import sys
 2import time
 3import queue
 4
 5from pydevdtk.coms.ble import Ble
 6
 7
 8class Parser:
 9    def __init__(self):
10        self.buffer = b""
11
12    def parse_data(self, data):
13        self.buffer += data
14        lines = self.buffer.split(b"\n")
15        self.buffer = lines.pop()
16
17        for line in lines:
18            line = line.strip()
19            if line:
20                values = line.decode("ascii").split(",")
21                try:
22                    parsed_values = tuple(map(float, values))
23                    yield parsed_values
24                except ValueError:
25                    pass
26
27
28ble = Ble()
29print("Scanning for BLE devices")
30ble.start_scan()
31time.sleep(2)
32ble.stop_scan()
33
34ble_devs = ble.get_found_devices()
35ble_devs = [ble_dev for ble_dev in ble_devs if len(str(ble_dev)) > 0]
36ble_devs_names = [
37    str(ble_dev) for ble_dev in ble_devs if len(str(ble_dev)) > 0
38]
39
40print(f"Found devs: {ble_devs_names}\n")
41
42dev_name = input("Type in the BLE dev name to read the data from: ")
43if dev_name not in ble_devs_names:
44    print(f"Requested device {dev_name} not found")
45    sys.exit(1)
46
47# wait for connection to establish
48dev = ble_devs[ble_devs_names.index(dev_name)]
49print(f"Connecting to {dev}...")
50t_timeout = 5
51ble.connect(dev)
52t_start = time.time()
53while time.time() - t_start < t_timeout:
54    if ble.is_connected(dev):
55        break
56    else:
57        time.sleep(0.2)
58if not ble.is_connected(dev):
59    print(f"Could not connect to device {dev}")
60    sys.exit(1)
61
62print(f"Connected to {dev}")
63
64data_queue = queue.Queue()
65parser = Parser()
66ble.start_notifications(
67    dev,
68    "beb5483e-36e1-4688-b7f5-ea07361b26a8",
69    lambda data: data_queue.put(data),
70)
71print("Notifications started")
72
73t_run = 10
74t_start = time.time()
75while time.time() - t_start < t_run:
76    data = data_queue.get()
77    for sin, cos in parser.parse_data(data):
78        print(f"sin = {sin} | cos = {cos}")
79
80ble.stop_notifications(
81    dev,
82    "beb5483e-36e1-4688-b7f5-ea07361b26a8",
83)
84print("Notifications stopped")
85
86print(f"Disconnecting from {dev}...")
87ble.disconnect(dev)
88t_timeout = 5
89t_start = time.time()
90while time.time() - t_start < t_timeout:
91    if not ble.is_connected(dev):
92        break
93if ble.is_connected(dev):
94    print(f"Could not disconnect from device {dev}")
95    sys.exit(2)
96print(f"Disconnected from {dev}")

The plotter example demonstrates how to perform real-time plotting using plotting module. It simulates data and utilizes this data to create few real-time plots, including line plots, scatter plots, bar plots and one spectrogram plot using image plot.

Example code:

  1import multiprocessing as mp
  2import time
  3
  4import numpy as np
  5from scipy import fft
  6
  7from pydevdtk.plotting import Plotter, PlotterManager

 10def get_spectrum(fs, x, n_fft=None, log=False):
 11    n = x.size
 12    if n_fft is None:
 13        n_fft = int(2 ** np.ceil(np.log2(n)))
 14    x_spec = fft.fft(x, n_fft)
 15    x_spec = np.abs(x_spec)
 16    x_spec = x_spec / n
 17    n_keep = n_fft // 2 + 1
 18    x_spec = x_spec[0:n_keep]
 19    x_spec[1:-1] = 2 * x_spec[1:-1]
 20    f = np.linspace(0, fs / 2, n_keep)
 21    if log:
 22        eps = 1e-10
 23        x_spec[x_spec < eps] = eps
 24        x_spec = 20 * np.log10(x_spec)
 25
 26    return f, x_spec
 27
 28
 29def generate_sine_signal(
 30    num_samples, amplitude, frequency, sample_rate, offset=0, phase=0
 31):
 32    if sample_rate <= 0:
 33        raise ValueError("Sample rate can't be negative or zero")
 34    if frequency <= 0:
 35        raise ValueError("Frequency can't be negative or zero")
 36    if frequency > sample_rate / 2:
 37        raise ValueError("Frequency is greater than sample_rate/2")
 38    time = np.arange(num_samples) / sample_rate
 39    signal = amplitude * np.sin(2 * np.pi * frequency * time + phase) + offset
 40    return signal
 41
 42
 43def generate_sinesweep(f_start, f_stop, amplitude, t_duration, fs):
 44    phi = 0
 45    f = f_start
 46    delta = 2 * np.pi * f / fs
 47    f_delta = (f_stop - f_start) / (fs * t_duration)
 48    samples = []
 49    for _ in range(int(fs * t_duration)):
 50        sample = amplitude * np.sin(phi)
 51        phi += delta
 52        f += f_delta
 53        delta = 2 * np.pi * f / fs
 54        samples.append(sample)
 55    return samples
 56
 57
 58def main():
 59    fs = 50
 60    num_samples = 50
 61    sin_wave = generate_sine_signal(num_samples, 1, 5, fs)
 62    cos_wave = generate_sine_signal(num_samples, 1, 2, fs, 0, np.pi / 2)
 63    sinesweep_wave = generate_sinesweep(0.1, 10, 1, 5, num_samples)
 64    i_sin = 0
 65    i_cos = 0
 66    i_sinswp = 0
 67    n_fft = 64
 68    n_spec_update = 1
 69    spec_buf = np.zeros(n_fft)
 70    spectrogram = np.full(
 71        (n_fft // 2 + 1, num_samples // n_spec_update), np.nan
 72    )
 73    i_spec_buf = 0
 74    i_spectrogram = 0
 75    plotter_manager = PlotterManager(Plotter())
 76    plotter_manager.create_figure("fig_demo", 3, 6, figsize=(12, 7))
 77    plotter_manager.create_axis("ax_line1", "fig_demo", 0, 0, 1, 2)
 78    plotter_manager.create_axis("ax_line2", "fig_demo", 1, 0, 1, 2)
 79    plotter_manager.create_axis("ax_bar", "fig_demo", 2, 0, 1, 2)
 80    plotter_manager.create_axis("ax_scat", "fig_demo", 0, 2, 2, 2)
 81    plotter_manager.create_axis("ax_line3", "fig_demo", 2, 2, 1, 2)
 82    plotter_manager.create_axis("ax_line4", "fig_demo", 2, 4, 1, 2)
 83    plotter_manager.create_axis("ax_img", "fig_demo", 0, 4, 2, 2)

 86    plotter_manager.create_line_plot("line2", "ax_line2", num_samples)

 88    plotter_manager.create_line_plot(
 89        "line3_1", "ax_line3", num_samples, label="sin"
 90    )

 93    )

 95    plotter_manager.create_line_plot("line4", "ax_line4", num_samples)
 96    plotter_manager.create_image_plot(
 97        "img",
 98        "ax_img",
 99        (n_fft // 2 + 1, num_samples // n_spec_update),
100        cbar=True,
101        vmin=0,
102        vmax=1,
103        aspect="auto",
104        origin="lower",
105    )
106
107    plotter_manager.modify_axis(
108        "ax_line1",
109        [0, num_samples - 1],
110        [-1.1, 1.1],
111        ylabel="Sine",
112        yticks=[-1, 0, 1],
113    )
114    plotter_manager.modify_axis(
115        "ax_line2",
116        [0, num_samples - 1],
117        [-1.1, 1.1],
118        ylabel="Cosine",
119        yticks=[-1, 0, 1],
120    )
121    plotter_manager.modify_axis(
122        "ax_bar",
123        [-0.5, 1.5],
124        [-1.1, 1.1],
125        ylabel="Amplitudes",
126        xticks=[0, 1],
127        xticklabels=["Sine", "Cosine"],
128    )
129    plotter_manager.modify_axis(
130        "ax_scat",
131        [-1.1, 1.1],
132        [-1.1, 1.1],
133        xticks=[-1, 0, 1],
134        yticks=[-1, 0, 1],
135    )
136    plotter_manager.modify_axis(
137        "ax_line3",
138        [0, num_samples - 1],
139        [-1.1, 1.1],
140        xlabel="SinCos",
141        yticks=[-1, 0, 1],

156            "line1": sin_wave[i_sin],
157            "line2": cos_wave[i_cos],
158            "bar": [sin_wave[i_sin], cos_wave[i_cos]],
159            "line3_1": sin_wave[i_sin],
160            "line3_2": cos_wave[i_cos],
161            "scat": [sin_wave[i_sin], cos_wave[i_cos]],
162            "line4": sinesweep_wave[i_sinswp],
163        }
164        spec_buf[i_spec_buf] = sinesweep_wave[i_sinswp]
165        if i_spec_buf % n_spec_update == 0:
166            _, spec = get_spectrum(fs, spec_buf, n_fft)
167            if i_spectrogram == spectrogram.shape[1] - 1:
168                spectrogram[:, : spectrogram.shape[1] - 1] = spectrogram[:, 1:]
169                spectrogram[:, i_spectrogram] = spec
170            else:
171                spectrogram[:, i_spectrogram] = spec
172                i_spectrogram += 1
173            data["img"] = spectrogram.copy()
174        plotter_manager.add_data(data)
175        i_sin += 1
176        i_cos += 1
177        i_sinswp += 1
178        i_spec_buf += 1
179        if i_sin >= len(sin_wave):
180            i_sin = 0
181        if i_cos >= len(cos_wave):
182            i_cos = 0
183        if i_sinswp >= len(sinesweep_wave):
184            i_sinswp = 0
185        if i_spec_buf >= num_samples:
186            i_spec_buf -= n_spec_update
187            spec_buf[: len(spec_buf) - n_spec_update] = spec_buf[
188                n_spec_update:
189            ]
190        time.sleep(1 / 50)
191    plotter_manager.stop()
192

195    mp.freeze_support()

Custom plotterÔÉÅ

The custom plotter example demonstrates how to subclass PlotterBase to have flexible figure and artists creation, using matplotlib APIs. It simulates data and utilizes this data to for few real-time plots.

Example code:

  1import multiprocessing as mp
  2import time
  3
  4import numpy as np
  5import matplotlib.pyplot as plt
  6
  7from pydevdtk.plotting import PlotterManager, PlotterBase
  8
  9
 10def generate_sine_signal(
 11    num_samples, amplitude, frequency, sample_rate, offset=0, phase=0
 12):
 13    if sample_rate <= 0:
 14        raise ValueError("Sample rate can't be negative or zero")
 15    if frequency <= 0:
 16        raise ValueError("Frequency can't be negative or zero")
 17    if frequency > sample_rate / 2:
 18        raise ValueError("Frequency is greater than sample_rate/2")
 19    time = np.arange(num_samples) / sample_rate
 20    signal = amplitude * np.sin(2 * np.pi * frequency * time + phase) + offset
 21    return signal
 22
 23
 24# create custom PlotterBase
 25class Plotter(PlotterBase):
 26    def __init__(self, num_points=50):
 27        self.num_points = num_points
 28
 29    def init(self):
 30        fig, axs = plt.subplots(3, 1, figsize=(8, 5), constrained_layout=True)
 31
 32        self.y_sin = self.num_points * [np.nan]
 33        self.y_cos = self.num_points * [np.nan]
 34        self.y_rand = self.num_points * [np.nan]
 35
 36        # create artists
 37        self.sin_artist = axs[0].plot(self.y_sin, color="C0")[0]
 38        self.cos_artist = axs[1].plot(self.y_cos, color="C1")[0]
 39        self.rand_artist = axs[2].plot(self.y_rand, color="C2")[0]
 40
 41        # modify axes properties
 42        for ax in axs:
 43            ax.set_xlim([0, self.num_points - 1])
 44        for ax in axs:
 45            ax.set_xticks([0, self.num_points / 2, self.num_points])
 46        axs[0].set_title("sin")
 47        axs[1].set_title("cos")
 48        axs[2].set_title("rand")
 49        axs[0].set_ylim([-1.1, 1.1])
 50        axs[1].set_ylim([-1.1, 1.1])
 51        axs[2].set_ylim([999, 5001])
 52
 53        # ensure this is called!
 54        self.add_figure_and_artists(
 55            fig, [self.sin_artist, self.cos_artist, self.rand_artist]
 56        )
 57
 58    def process_data_queue(self):
 59        while not self.data_queue.empty():
 60            sin, cos, rand = self.data_queue.get()
 61            self.y_sin.append(sin)
 62            self.y_cos.append(cos)
 63            self.y_rand.append(rand)
 64            self.y_sin = self.y_sin[-self.num_points :]
 65            self.y_cos = self.y_cos[-self.num_points :]
 66            self.y_rand = self.y_rand[-self.num_points :]
 67            self.sin_artist.set_ydata(self.y_sin)
 68            self.cos_artist.set_ydata(self.y_cos)
 69            self.rand_artist.set_ydata(self.y_rand)
 70
 71
 72def main():
 73    fs = 50
 74    num_samples = 100
 75    sin_wave = generate_sine_signal(num_samples, 1, 5, fs)
 76    cos_wave = generate_sine_signal(num_samples, 1, 2, fs, 0, np.pi / 2)
 77    i_sin = 0
 78    i_cos = 0
 79    plotter_manager = PlotterManager(Plotter(num_samples))
 80    plotter_manager.show()
 81    while plotter_manager.is_shown():
 82        data = [
 83            sin_wave[i_sin],
 84            cos_wave[i_cos],
 85            np.random.randint(1000, 5000 + 1),
 86        ]
 87        plotter_manager.add_data(data)
 88        i_sin += 1
 89        i_cos += 1
 90        if i_sin >= len(sin_wave):
 91            i_sin = 0
 92        if i_cos >= len(cos_wave):
 93            i_cos = 0
 94        time.sleep(1 / 50)
 95    plotter_manager.stop()
 96
 97
 98if __name__ == "__main__":
 99    mp.freeze_support()
100    main()
